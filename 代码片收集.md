- **JavaScript-随机数的生成**
  从一组有序的数据中生成一组随机并且不重复的数，类似于简单的抽奖程序的实现。
```JavaScript
var gRandomArr = function (arr, length) {
    // 使用sort将原数组的顺序打乱，让有序变成无序
    arr.sort(function () {
        return Math.random() - 0.5;
    });

    // 从原数组中一次性返回10个元素
    return arr.slice(0, length);
};
// 调用
gRandomArr(arr, 10);
```

- **JavaScript-限制“Text-Area”域中的字符的个数**
```JavaScript
jQuery.fn.maxLength = function (max) {
	this.each(function () {
		var type = this.tagName.toLowerCase();
		var inputType = this.type ? this.type.toLowerCase() : null;
		if (type == "input" && inputType == "text" || inputType == "password") {
			this.maxLength = max;
		} else if (type == "textarea") {
			this.onkeypress = function (e) {
				var ob = e || event;
				var keyCode = ob.keyCode;
				var hasSelection = document.selection
					 ? document.selection.createRange().text.length > 0
					 : this.selectionStart != this.selectionEnd;
				return !(this.value.length >= max
					 && (keyCode > 50 || keyCode == 32 || keyCode == 0 || keyCode == 13)
					 && !ob.ctrlKey && !ob.altKey && !hasSelection);
			};
			this.onkeyup = function () {
				if (this.value.length > max) {
					this.value = this.value.substring(0, max);
				}
			};
		}
	});
};
//用法 $('#mytextarea').maxLength(500);
```

- **从元素中除去HTML**

(function ($) {
	$.fn.stripHtml = function () {
		var regexp = /<("[^"]*"|'[^']*'|[^'">])*>/gi;
		this.each(function () {
			$(this).html($(this).html().replace(regexp, ""));
		});
		return $(this);
	}
})(jQuery);
//用法： $('p').stripHtml();

// 半角转全角
public static string ToSBC(this string input)
        {
            char[] c = input.ToCharArray();
            for (int i = 0; i < c.Length; i++)
            {
                if (c[i] == 32)
                {
                    c[i] = (char)12288;
                    continue;
                }
                if (c[i] < 127)
                    c[i] = (char)(c[i] + 65248);
            }
            return new string(c);
        }
//全角转半角
public static string ToDBC(this string input)
        {
            char[] c = input.ToCharArray();
            for (int i = 0; i < c.Length; i++)
            {
                if (c[i] == 12288)
                {
                    c[i] = (char)32;
                    continue;
                }
                if (c[i] > 65280 && c[i] < 65375)
                    c[i] = (char)(c[i] - 65248);
            }
            return new string(c);
        }
